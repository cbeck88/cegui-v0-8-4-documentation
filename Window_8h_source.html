97  {

99  if (str == "Always")

102  }
103  else if (str == "Never")
104  {
105  return WUM_NEVER;
106  }
107  else

109  return WUM_VISIBLE;

117  return "Always";

119  else if (val == WUM_NEVER)

121  return "Never";
122  }
123  else if (val == WUM_VISIBLE)
124  {
125  return "Visible";
126  }
127  else
128  {
129  assert(false && "Invalid Window Update Mode");

131  }
